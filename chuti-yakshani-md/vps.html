DEPLOY ON VPS

sudo apt update
sudo apt install git
sudo apt install nodejs npm
git clone https://github.com/mrdilla/Chuti-Yakshani-MD.git
cd Chuti_Yakshani-MD
npm install
nano ~/.bash_profile
export SESSION_ID="enter session-id"
export GITHUB_AUTH_TOKEN="enter github auth token"
export GITHUB_USERNAME="enter github username"
export BOT_NUMBER="enter botnumber"
                
source ~/.bash_profile
npm start
